CREATE TABLE estado(
	id SERIAL PRIMARY KEY,
	estado_uf VARCHAR(2)
);

CREATE TABLE est_civil(
	id SERIAL PRIMARY KEY,
	est_civil VARCHAR(15)
);

CREATE TABLE cliente(
	id SERIAL PRIMARY KEY,
	nome VARCHAR(100),
	endereco VARCHAR(100),
	numero VARCHAR(5),
	bairro VARCHAR(100),
	cidade VARCHAR(100),
	estado_id INTEGER REFERENCES estado(id),
	cep VARCHAR(9),
	telefone VARCHAR(13),
	celular VARCHAR(14),
	email VARCHAR(100),
	data DATE,
	est_civil_id INTEGER REFERENCES est_civil(id)	
);
